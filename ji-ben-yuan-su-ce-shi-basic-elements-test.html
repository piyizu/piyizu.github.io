2.2.3 Definition of Finite State Machine

A finite state machine of a program property is a quintuple where - is a finite and non-empty set of states of a program object - is a finite and non-empty set of program locations - is a function for state transition . The expression indicates that the state can transit to the state if the object goes through a program location where - is the initial state - is the set of finite states

import java.util.Scanner;
import java.util.Random;
public class Main {
    private static void quickSort(int[] a, int l, int r) {
        if(l < r) {
            int x = a[(new Random(114514)).nextInt(r - l) + l];  // random pivot for better performance
            int i = l - 1, j = r + 1;
            while (i < j) {
                do ++i; while (a[i] < x);  // strictly less (no equality), otherwise error may occur
                do --j; while (a[j] > x);  // same as above
                if (i < j) {
                    int temp = a[i];
                    a[i] = a[j];
                    a[j] = temp;
                }
            }
            quickSort(a, l, j);
            quickSort(a, j + 1, r);
        }
    }
    public static void main(String[] args) {
        int n;
        int[] ar = null;
        Scanner scanner = new Scanner(System.in);
        n = scanner.nextInt();
        ar = new int[n];
        for(int i = 0; i < n; ++i) {
            ar[i] = scanner.nextInt();
        }

        quickSort(ar, 0, n - 1);

        for(int x : ar) {
            System.out.printf("%d ", x);
        }
    }
}
